```java
class Solution {
    // 递归法
    public int sumNums(int n) {
        // 将判断是否为递归的条件看作A,将递归的主体函数看作B
        // 即在 A && B 中,只有 A 为 true 时才执行B哟
        boolean flag = n > 0 && (n += sumNums(n - 1)) > 0;
        return n;
    }
}
```

```python
'''

class Solution(object):
    def sumNums(self, n):
        """
        :type n: int
        :rtype: int
        """
        return sum(range(n + 1))
```
